%
% Demo doc that tests most of the features of the cu-thesis class
% This version tests the 'draft' option, which generates a editor-optimized 
% thesis that has some extra features for editing (does not fully comply to Carleton specs)
%
% NOTE: if cu-thesis is installed on your system, documentclass can just be cu-thesis
%
\documentclass[
    draft,
    scrbook,
    listoffigures,
    listoftables,
    glossary]{../cu-thesis}

% Packages for this test doc to generate text and verify figures
% you probably don't want these
\usepackage{blindtext}
\usepackage{lipsum}

% Set up bibtex or biblatex here
\usepackage{cite}

% Layout is only used to show the page layout in this demo doc; you don't need this for a real thesis
\usepackage{layout}

\addtokomafont{disposition}{\rmfamily}

% SETUP: Document Details
\title{Thesis Title}
\author{Author Name}
\date{January 1970}

% The title page uses the following options to piece together the subject line 
% "A [thesistype] submitted to [submitted to] in partial fullfilment of ... etc
\thesistype{thesis proposal}
\submittedto{the thesis proposal committee}

%% To specify a degree and program if the degree and program are different
% \degree{Master of Something}
% \program{Degree Program}

% You can also use \degreeprogram to specify your degree and program together
% this is useful if your degree and program are the same
% E.g., to avoid "Master's of Computer Science in Computer Science"
%   you can use \degreeprogram{Master's of Computer Science}
% this overrides \degree and \program
\degreeprogram{Master of Something}

\submissiondate{2022}

% Abstract, acknowledgements
\abstract{\lipsum[1]}
\acknowledgements{\lipsum[1]}

% PDF meta-info:
\subjectline{\@title}
\keywords{keyword 1, keyword 2, etc.}

\begin{document}
    % Create the frontmatter (title page, abstract, toc, etc.)
    \frontmatter

    % Create sample document
    \Blinddocument
    Example citation~\cite{nash51}

    % Show layout for debug purposes 
    \layout
    
    \appendix

    % Bibliography
    \bibliographystyle{unsrt}
    \bibliography{test-bib}
\end{document}
